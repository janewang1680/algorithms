827. Making A Large Island

Hard

In a 2D grid of 0s and 1s, we change at most one 0 to a 1.

After, what is the size of the largest island? (An island is a 4-directionally connected group of 1s).

Example 1:

Input: [[1, 0], [0, 1]]
Output: 3
Explanation: Change one 0 to 1 and connect two 1s, then we get an island with area = 3.

Example 2:

Input: [[1, 1], [1, 0]]
Output: 4
Explanation: Change the 0 to 1 and make the island bigger, only one island with area = 4.

Example 3:

Input: [[1, 1], [1, 1]]
Output: 4
Explanation: Can’t change any 0 to 1, only one island with area = 4.

Notes:

  • 1 <= grid.length = grid[0].length <= 50.
  • 0 <= grid[i][j] <= 1.

[Solution]

class Solution {
    public int largestIsland(int[][] grid) {
        if (grid == null || grid.length == 0 || grid[0].length == 0) return 0;
        
        int m = grid.length, n = grid[0].length;
        
        Map<Integer, Integer> areas = new HashMap<>();
        Map<Integer, Set<Integer>> map = new HashMap<>();
        int cnt = 2;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    int area = dfs(grid, i, j, cnt, map);
                    areas.put(cnt++, area);
                }
            }
        }
        
        int maxArea = 1;
        for (int area : areas.values()) {
            maxArea = Math.max(maxArea, area);
        }
        
        for (Integer idx : map.keySet()) {
            int area = 1;
            for (int island: map.get(idx)) {
                area += areas.get(island);
            }
            maxArea = Math.max(maxArea, area);
        }
        
        return maxArea;
    }
    
    int dfs(int[][] grid, int row, int col, int cnt, Map<Integer, Set<Integer>> map) {
        if (row < 0 || col < 0 || row >= grid.length || col >= grid[0].length) return 0;
        if (grid[row][col] > 1) return 0;
        if (grid[row][col] == 0) {
            int idx = row * grid[0].length + col;
            map.computeIfAbsent(idx, z-> new HashSet<>()).add(cnt);
            return 0;
        }
        
        grid[row][col] = cnt;
        
        int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        
        int area = 1;
        for (int i = 0; i < 4; i++) {
            int x = row + dirs[i][0], y = col + dirs[i][1];
            area += dfs(grid, x, y, cnt, map);
        }
        
        return area;
    }
}
