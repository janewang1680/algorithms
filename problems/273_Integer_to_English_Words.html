273. Integer to English Words

Hard

Convert a non-negative integer num to its English words representation.

Example 1:

Input: num = 123
Output: “One Hundred Twenty Three”

Example 2:

Input: num = 12345
Output: “Twelve Thousand Three Hundred Forty Five”

Example 3:

Input: num = 1234567
Output: “One Million Two Hundred Thirty Four Thousand Five Hundred Sixty Seven”

Example 4:

Input: num = 1234567891
Output: “One Billion Two Hundred Thirty Four Million Five Hundred Sixty Seven Thousand Eight Hundred Ninety One”

Constraints:

  • 0 <= num <= 231 - 1

[Solution]

class Solution {
    public String numberToWords(int num) {
        if (num == 0) return "Zero";
        
        String[] thousandStrs = {"Billion", "Million", "Thousand", ""};
        int[] thousands = {1000000000, 1000000, 1000, 1};
        
        StringBuilder builder = new StringBuilder();
        for (int i = 0; i < 4; i++) {
            int n = num / thousands[i];
            if (n > 0) {
                if (builder.length() > 0) builder.append(" ");
                builder.append(toString(n));
                if (thousandStrs[i].length() > 0) builder.append(" " + thousandStrs[i]);
            }
            
            num = num % thousands[i];
        }
        
        return builder.toString();
    }
    
    String toString(int n) {
        Map<Integer, String> map = new HashMap<>();
        map.put(1, "One");
        map.put(2, "Two");
        map.put(3, "Three");
        map.put(4, "Four");
        map.put(5, "Five");
        map.put(6, "Six");
        map.put(7, "Seven");
        map.put(8, "Eight");
        map.put(9, "Nine");
        map.put(10, "Ten");
        map.put(11, "Eleven");
        map.put(12, "Twelve");
        map.put(13, "Thirteen");
        map.put(14, "Fourteen");
        map.put(15, "Fifteen");
        map.put(16, "Sixteen");
        map.put(17, "Seventeen");
        map.put(18, "Eighteen");
        map.put(19, "Nineteen");
        map.put(20, "Twenty");
        map.put(30, "Thirty");
        map.put(40, "Forty");
        map.put(50, "Fifty");
        map.put(60, "Sixty");
        map.put(70, "Seventy");
        map.put(80, "Eighty");
        map.put(90, "Ninety");
        
        StringBuilder builder = new StringBuilder();
        if (n / 100 > 0) {
            builder.append(map.get(n / 100));
            builder.append(" " + "Hundred");
        }
        n = n % 100;
        if (n == 0) return builder.toString();
        if (builder.length() > 0) builder.append(" ");
        
        if (n <= 20) {
            builder.append(map.get(n));
        } else {
            int d = n / 10;
            builder.append(map.get(d * 10));
            
            n = n % 10;
            if (n > 0) {
                builder.append(" " + map.get(n));
            }
        }
        
        return builder.toString();
    }
}
