336. Palindrome Pairs

Hard

Given a list of unique words, return all the pairs of the distinct indices (i, j) in the given list, so that the concatenation of the two words words[i] + words[j] is a palindrome.

Example 1:

Input: words = [“abcd”,“dcba”,“lls”,“s”,“sssll”]
Output: [[0,1],[1,0],[3,2],[2,4]]
Explanation: The palindromes are [“dcbaabcd”,“abcddcba”,“slls”,“llssssll”]

Example 2:

Input: words = [“bat”,“tab”,“cat”]
Output: [[0,1],[1,0]]
Explanation: The palindromes are [“battab”,“tabbat”]

Example 3:

Input: words = [“a”,""]
Output: [[0,1],[1,0]]

Constraints:

  • 1 <= words.length <= 5000
  • 0 <= words[i].length <= 300
  • words[i] consists of lower-case English letters.

[Solution]

class Solution {
    public List<List<Integer>> palindromePairs(String[] words) {
        List<List<Integer>> res = new ArrayList<>();
        Map<String, Integer> dict = new HashMap<>();
        for (int i = 0; i < words.length; i++) dict.put(words[i], i);
        
        for (int i = 0; i < words.length; i++) {
            if (isPalindrome(words[i])) {
                if (dict.containsKey("") && !words[i].equals("")) {
                    List<Integer> list = new ArrayList<>();
                    list.add(i);
                    list.add(dict.get(""));
                    res.add(list);
                    
                    list = new ArrayList<>();
                    list.add(dict.get(""));
                    list.add(i);
                    res.add(list);
                }
            }
                
            String rev = reverse(words[i]);
            if (dict.containsKey(rev) && !rev.equals(words[i])) {
                List<Integer> list = new ArrayList<>();
                list.add(i);
                list.add(dict.get(rev));
                res.add(list);
            }
            
            for (int j = 1; j < words[i].length(); j++) {
                String first = words[i].substring(0, j);
                String second = words[i].substring(j);
                
                if (isPalindrome(first)) {
                    rev = reverse(second);
                    if (dict.containsKey(rev)) {
                        List<Integer> list = new ArrayList<>();
                        list.add(dict.get(rev));
                        list.add(i);
                        res.add(list);
                    }
                }
                
                if (isPalindrome(second)) {
                    rev = reverse(first);
                    if (dict.containsKey(rev)) {
                        List<Integer> list = new ArrayList<>();
                        list.add(i);
                        list.add(dict.get(rev));                        
                        res.add(list);
                    }
                }
            }    
            
        }
        
        return res;
    }
    
    boolean isPalindrome(String str) {
        int i = 0, j= str.length() - 1;
        while (i < j) {
            if (str.charAt(i++) != str.charAt(j--)) return false;
        }
        
        return true;
    }
                
    String reverse(String str) {
        StringBuilder builder = new StringBuilder(str);
        return builder.reverse().toString();
    }
}
