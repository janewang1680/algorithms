126. Word Ladder II

Hard

Given two words (beginWord and endWord), and a dictionary’s word list, find all shortest transformation sequence(s) from beginWord to endWord, such that:

  1. Only one letter can be changed at a time
  2. Each transformed word must exist in the word list. Note that beginWord is not a transformed word.

Note:

  • Return an empty list if there is no such transformation sequence.
  • All words have the same length.
  • All words contain only lowercase alphabetic characters.
  • You may assume no duplicates in the word list.
  • You may assume beginWord and endWord are non-empty and are not the same.

Example 1:

Input:
beginWord = “hit”,
endWord = “cog”,
wordList = [“hot”,“dot”,“dog”,“lot”,“log”,“cog”]

Output:
[
[“hit”,“hot”,“dot”,“dog”,“cog”],
[“hit”,“hot”,“lot”,“log”,“cog”]
]

Example 2:

Input:
beginWord = “hit”
endWord = “cog”
wordList = [“hot”,“dot”,“dog”,“lot”,“log”]

Output: []

Explanation: The endWord “cog” is not in wordList, therefore no possible transformation.

[Solution]

class Solution {
    static class Node {
        String word;
        Set<Node> prevs;
        int step;
        
        public Node(String word) {
            this.word = word;
            this.prevs = new HashSet<>();
            this.step = 0;
        }
    }
    
    
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        
        List<List<String>> res = new ArrayList<>();
        
        Set<String> dict = new HashSet<>(wordList);
        if (!dict.contains(endWord)) return res;
        
        Queue<Node> q = new LinkedList<>();
        q.offer(new Node(beginWord));
        Map<String, Node> visited = new HashMap<>();
        visited.put(beginWord, q.peek());
        
        int len = beginWord.length();
        
        int minStep = -1;
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                
                Node node = q.poll();
                String word = node.word;
                if (word.equals(endWord)) {
                    if (minStep == -1) minStep = node.step;
                    else if (node.step > minStep) break;
                }
                
                for (int j = 0; j < len; j++) {
                    char c = word.charAt(j);
                    for (int k = 0; k < 26; k++) {
                        char d = (char) (k + 'a');
                        if (c == d) continue;
                        String tmp = node.word.substring(0, j) + d + node.word.substring(j+1);
                        if (!dict.contains(tmp)) continue;
                        if (visited.containsKey(tmp)) {
                            Node tmpNode = visited.get(tmp);
                            if (tmpNode.step == node.step + 1) {
                                tmpNode.prevs.add(node);
                            }
                        } else {
                            Node tmpNode = new Node(tmp);
                            tmpNode.prevs.add(node);
                            tmpNode.step = node.step + 1;
                            visited.put(tmp, tmpNode);
                            q.offer(tmpNode);
                        }
                    }
                }
            }
        }
        
        if (minStep == -1) return res;
        else {
            backTrace(visited, endWord, beginWord, new ArrayList<String>(), res);
            return res;
        }
    }

                    for (int k = 0; k < 25; k++) {
                        char d = (char)(k + 'a');
                        if (c == d) continue;
                        String tmp = word.substring(0, j) + d + word.substring(j+1);
                        if (!dict.contains(tmp)) continue;
                        if(visited.containsKey(tmp)) {
                            Node tmpNode = visited.get(tmp);
                            if (tmpNode.step == node.step+1) {
                                tmpNode.prevs.add(node);
                            }
                        } else {
                            visited.put(tmp, new Node(tmp));
                            visited.get(tmp).step = node.step+1;
                            visited.get(tmp).prevs.add(node);
                            q.offer(visited.get(tmp));
                        }
                    }
                }
            }
        }
        
        if (minStep == -1) return res;
        backTrace(visited, endWord, beginWord, new ArrayList<String>(), res);
        
        return res;
    }
    
    void backTrace(Map<String, Node> map, String beginWord, String endWord, List<String> path, List<List<String>> res) {
        if (beginWord.equals(endWord)) {
            List<String> list = new ArrayList<>(path);
            list.add(endWord);
            Collections.reverse(list);
            res.add(list);
            return;
        }
        
        path.add(beginWord);
        
        for (Node node: map.get(beginWord).prevs) {
            backTrace(map, node.word, endWord, path, res);
        }
        
        path.remove(path.size() - 1);
    }    
}
