1245. Tree Diameter

Medium

Given an undirected tree, return its diameter: the number of edges in a longest path in that tree.

The tree is given as an array of edges where edges[i] = [u, v] is a bidirectional edge between nodes u and v. Each node has labels in the set {0, 1, ..., edges.length}.

Example 1:

Input: edges = [[0,1],[0,2]]
Output: 2
Explanation:
A longest path of the tree is the path 1 - 0 - 2.

Example 2:

Input: edges = [[0,1],[1,2],[2,3],[1,4],[4,5]]
Output: 4
Explanation:
A longest path of the tree is the path 3 - 2 - 1 - 4 - 5.

Constraints:

  • 0 <= edges.length < 10^4
  • edges[i][0] != edges[i][1]
  • 0 <= edges[i][j] <= edges.length
  • The given edges form an undirected tree.

[Solution]

class Solution {
    public int treeDiameter(int[][] edges) {
        if (edges == null || edges.length == 0) return 0;
        
        int n = edges.length;
        Set<Integer>[] tree = new Set[n+1];
        for (int i = 0; i < n; i++) {
            int n1 = edges[i][0];
            int n2 = edges[i][1];
            if (tree[n1] == null) tree[n1] = new HashSet<>();
            if (tree[n2] == null) tree[n2] = new HashSet<>();
            tree[n1].add(n2);
            tree[n2].add(n1);
        }
        
        int[] res = helper(tree, 0, -1);
        return res[0] - 1;
    }
    
    int[] helper(Set<Integer>[] tree, int root, int prev) {
        
        int d = 0, max1 = 0, max2 = 0;
        
        for (int node: tree[root]) {
            if (node == prev) continue;
            int[] res = helper(tree, node, root);
            d = Math.max(d, res[0]);
            if (res[1] >= max1) {
                max2 = max1;
                max1 = res[1];
            } else if (res[1] >= max2) {
                max2 = res[1];
            }
        }
        
        d = Math.max(1 + max1 + max2, d);
        return new int[]{d, 1 + max1};
    }
}
