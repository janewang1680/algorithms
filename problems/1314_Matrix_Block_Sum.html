1314. Matrix Block Sum

Medium

Given a m * n matrix mat and an integer K, return a matrix answer where each answer[i][j] is the sum of all elements mat[r][c] for i - K <= r <= i + K, j - K <= c <= j + K, and (r, c) is a valid position in the matrix.

Example 1:

Input: mat = [[1,2,3],[4,5,6],[7,8,9]], K = 1
Output: [[12,21,16],[27,45,33],[24,39,28]]

Example 2:

Input: mat = [[1,2,3],[4,5,6],[7,8,9]], K = 2
Output: [[45,45,45],[45,45,45],[45,45,45]]

Constraints:

  • m == mat.length
  • n == mat[i].length
  • 1 <= m, n, K <= 100
  • 1 <= mat[i][j] <= 100

[Solution]

class Solution {
    public int[][] matrixBlockSum(int[][] mat, int K) {
        int m = mat.length, n = mat[0].length;
        
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                mat[i][j] += mat[i-1][j];
            }
        }
        
        for (int j = 1; j < n; j++) {
            for (int i = 0; i < m; i++) {
                mat[i][j] += mat[i][j-1];
            }
        }
        
        
        int[][] res = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int minX = Math.max(0, i - K);
                int minY = Math.max(0, j - K);
                int maxX = Math.min(m - 1, i+K);
                int maxY = Math.min(n - 1, j + K);
                
                
                int sum = mat[maxX][maxY];
                if (minX > 0) sum -= mat[minX - 1][maxY];
                if (minY > 0) sum -= mat[maxX][minY - 1];
                if (minX > 0. && minY > 0) sum += mat[minX-1][minY-1];
                
                res[i][j] = sum;
            }
        }
        
        return res;
    }
}
