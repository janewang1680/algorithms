22. Generate Parentheses

Medium

Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

Example 1:

Input: n = 3
Output: ["((()))","(()())","(())()","()(())","()()()"]

Example 2:

Input: n = 1
Output: ["()"]

Constraints:

  • 1 <= n <= 8

[Solution]

class Solution {
    public List<String> generateParenthesis(int n) {
        List<String> res = new ArrayList<>();
        if (n == 0) return res;
        
        helper(n, 0, 0, "", res);
        return res;
    }
    
    void helper(int n, int left, int right, String aResult, List<String> res) {
        if (left == n && right == n) {
            res.add(aResult);
            return;
        }
        
        if (left < n) {
            helper(n, left+1, right, aResult+ "(", res);
        }
        if (right < n && left > right) {
            helper(n, left, right+1, aResult + ")", res);
        }
    }
}
