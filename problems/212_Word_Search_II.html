212. Word Search II

Hard

Given an m x n board of characters and a list of strings words, return all words on the board.

Each word must be constructed from letters of sequentially adjacent cells, where adjacent cells are horizontally or vertically neighboring. The same letter cell may not be used more than once in a word.

Example 1:

Input: board = [[“o”,“a”,“a”,“n”],[“e”,“t”,“a”,“e”],[“i”,“h”,“k”,“r”],[“i”,“f”,“l”,“v”]], words = [“oath”,“pea”,“eat”,“rain”]
Output: [“eat”,“oath”]

Example 2:

Input: board = [[“a”,“b”],[“c”,“d”]], words = [“abcb”]
Output: []

Constraints:

  • m == board.length
  • n == board[i].length
  • 1 <= m, n <= 12
  • board[i][j] is a lowercase English letter.
  • 1 <= words.length <= 3 * 104
  • 1 <= words[i].length <= 10
  • words[i] consists of lowercase English letters.
  • All the strings of words are unique.

[Solution]

class Solution {
    static class Trie {
        char c;
        Trie[] children;
        boolean isWord;
        
        public Trie(char c) {
            this.c = c;
            this.children = new Trie[26];
            this.isWord = false;
        }
    }
    
    public List<String> findWords(char[][] board, String[] words) {
        
        List<String> res = new ArrayList<>();
        if (board == null || board.length == 0 || board[0].length == 0 || words == null || words.length == 0) return res;
    
        Trie root = new Trie(' ');
        for (String word: words) addWord(word, root);
        
        Set<String> set = new HashSet<>();
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                boolean[][] visited = new boolean[board.length][board[0].length];
                dfs(board, root, i, j, "", visited, set);
            }
        }
        
        return new ArrayList<String>(set);
    }
    
    void dfs(char[][] board, Trie root, int row, int col, String tmp, boolean[][] visited, Set<String> set) {
        if (root.isWord) set.add(tmp);
        
        if (row < 0 || col < 0 || row >= board.length || col >= board[0].length) return;
        if (visited[row][col]) return;
    
        Trie node = root.children[board[row][col] - 'a'];
        if (node == null) {
            return;
        }
        
        visited[row][col] = true;
        
        int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for (int i = 0; i < 4; i++) {
            int x = row + dirs[i][0], y = col + dirs[i][1];
            dfs(board, node, x, y, tmp+board[row][col], visited, set);
        }
        
        visited[row][col] = false;
    }
    
    void addWord(String word, Trie root) {
        Trie node = root;
        for (int i = 0; i < word.length(); i++) {
            char c = word.charAt(i);
            if (node.children[c - 'a'] == null) {
                node.children[c - 'a'] = new Trie(c);
            }
            node = node.children[c - 'a'];
        }
        
        node.isWord = true;
    }
}
