247. Strobogrammatic Number II

Medium

A strobogrammatic number is a number that looks the same when rotated 180 degrees (looked at upside down).

Find all strobogrammatic numbers that are of length = n.

Example:

Input: n = 2
Output: ["11","69","88","96"]

[Solution]

class Solution {
    public List<String> findStrobogrammatic(int n) {
        List<String> res = new ArrayList<>();
        if (n < 1) return res;
        
        helper("", n, res);
        helper("0", n, res);
        helper("1", n, res);
        helper("8", n, res);
        
        return res;
    }
    
    void helper(String s, int n, List<String> res) {
        if (s.length() > n) return;
        if (s.length() == n) {
            if (s.charAt(0) != '0') res.add(s);
            else if (n == 1) res.add(s);
            return;
        }
        
        
        helper("0" + s + "0", n, res);
        helper("1" + s + "1", n, res);
        helper("6" + s + "9", n, res);
        helper("8" + s + "8", n, res);
        helper("9" + s + "6", n, res);
        
    }
}
