106. Construct Binary Tree from Inorder and Postorder Traversal

Medium

Given inorder and postorder traversal of a tree, construct the binary tree.

Note:
You may assume that duplicates do not exist in the tree.

For example, given

inorder = [9,3,15,20,7]
postorder = [9,15,7,20,3]

Return the following binary tree:

3

/
9 20
/
15 7

[Solution]

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode buildTree(int[] inorder, int[] postorder) {
         if (postorder == null || postorder.length == 0) return null;
        return buildTree(postorder, 0, postorder.length -1, inorder, 0, inorder.length - 1);
    }
    TreeNode buildTree(int[] postorder, int pl, int pu, int[] inorder, int il, int iu) {
        if (pl > pu) return null;
        TreeNode root = new TreeNode(postorder[pu]);
        if(pl == pu) return root;
        
        int j = il;
        while (j <= iu) {
            if (inorder[j] == postorder[pu]) break;
            j++;
        }
        
        root.left = buildTree(postorder, pl, j - il + pl-1, inorder, il, j-1);
        root.right = buildTree(postorder, j-il + pl, pu-1, inorder, j+1, iu);
        return root;
    }
}
