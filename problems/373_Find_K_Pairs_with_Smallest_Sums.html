373. Find K Pairs with Smallest Sums

Medium

You are given two integer arrays nums1 and nums2 sorted in ascending order and an integer k.

Define a pair (u,v) which consists of one element from the first array and one element from the second array.

Find the k pairs (u1,v1),(u2,v2) …(uk,vk) with the smallest sums.

Example 1:

Input: nums1 = [1,7,11], nums2 = [2,4,6], k = 3
Output: [[1,2],[1,4],[1,6]]
Explanation: The first 3 pairs are returned from the sequence:
[1,2],[1,4],[1,6],[7,2],[7,4],[11,2],[7,6],[11,4],[11,6]

Example 2:

Input: nums1 = [1,1,2], nums2 = [1,2,3], k = 2
Output: [1,1],[1,1] Explanation: The first 2 pairs are returned from the sequence:
[1,1],[1,1],[1,2],[2,1],[1,2],[2,2],[1,3],[1,3],[2,3]

Example 3:

Input: nums1 = [1,2], nums2 = [3], k = 3
Output: [1,3],[2,3] Explanation: All possible pairs are returned from the sequence: [1,3],[2,3]

[Solution]

class Solution {
    public List<List<Integer>> kSmallestPairs(int[] nums1, int[] nums2, int k) {
        List<List<Integer>> res = new ArrayList<>();
        k = Math.min(k, nums1.length * nums2.length);
        
        PriorityQueue<List<Integer>> pq = new PriorityQueue<>((a, b)->{
            int sum1 = nums1[a.get(0)] + nums2[a.get(1)];
            int sum2 = nums1[b.get(0)] + nums2[b.get(1)];
            return sum1 - sum2;
        });
        Set<List<Integer>> visited = new HashSet<>();
        
        List<Integer> list = new ArrayList<>();
        list.add(0);
        list.add(0);
        pq.offer(list);
        visited.add(list);
        
        while (k > 0) {
            List<Integer> cur = pq.poll();
            List<Integer> ares = new ArrayList<>();
            ares.add(nums1[cur.get(0)]);
            ares.add(nums2[cur.get(1)]);
            res.add(ares);
            k--;
            
            
            if (cur.get(0) < nums1.length -1) {
                List<Integer> list1 = new ArrayList<>();
                list1.add(cur.get(0)+1);
                list1.add(cur.get(1));
                if (!visited.contains(list1)) {
                    visited.add(list1);
                    pq.offer(list1);
                }
            }
            
            if (cur.get(1) < nums2.length -1) {
                List<Integer> list2 = new ArrayList<>();
                list2.add(cur.get(0));
                list2.add(cur.get(1)+1);
                if (!visited.contains(list2)) {
                    visited.add(list2);
                    pq.offer(list2);
                }
            }
        }
        
        return res;
    }
}
