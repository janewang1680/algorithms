59. Spiral Matrix II

Medium

Given a positive integer n, generate an n x n matrix filled with elements from 1 to n2 in spiral order.

Example 1:

Input: n = 3
Output: [[1,2,3],[8,9,4],[7,6,5]]

Example 2:

Input: n = 1
Output: [[1]]

Constraints:

  • 1 <= n <= 20

[Solution]

class Solution {
    public int[][] generateMatrix(int n) {
        int top = 0;
        int left = 0;
        int right = n - 1;
        int bottom = n - 1;
        
        int[][] res = new int[n][n];
        int num = 1;
        
        while (top <= bottom && left <= right) {
            
            for (int i = left; i <= right; i++) {
                res[top][i] = num++;
            }
            
            if (++top <= bottom) {
                for (int i = top; i <= bottom; i++) {
                    res[i][right] = num++;
                }
            }
            
            if (--right >= left) {
                for (int i = right; i >= left; i--) {
                    res[bottom][i] = num++;
                }
            }
            
            if (--bottom >= top) {
                for (int i = bottom; i >= top; i--) {
                    res[i][left] = num++;
                }
            }
            
            ++left;
        }
        
        return res;
    }
}
